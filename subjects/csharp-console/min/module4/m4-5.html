We splitsen de applicatie in twee statussen: Hoofdmenu (geen dier geselecteerd) en Actie Menu (ingelogd op een dier).

// In Menu.cs

public int AskInt(string question)
{
    Console.Write(question + " ");
    if (int.TryParse(Console.ReadLine(), out int result)) return result;
    return 50; // Fallback waarde
}

public void ShowPetMenu(string name, int energy, int hunger)
{
    Console.WriteLine($"\n--- MENU: {name} ---");
    Console.WriteLine($"Energie: {energy} | Honger: {hunger}");
    Console.WriteLine("0. Terug (Uitloggen)");
}

Stap 3: De Loop Splitsen (Game.cs)

Pas de Start() methode aan. De `if (_currentPet == null)` bepaalt welk menu getoond wordt.

public void Start()
{
    _menu.ShowTitle();
    bool appRunning = true;

    while (appRunning)
    {
        // SITUATIE A: HOOFDMENU
        if (_currentPet == null)
        {
            _menu.ShowMainMenu(_pets.Count);
            string choice = _menu.AskString("Keuze:");

            switch (choice.ToUpper())
            {
                case "1": // Nieuw
                    string name = _menu.AskString("Naam:");
                    int energy = _menu.AskInt("Energie (0-100):");
                    int hunger = _menu.AskInt("Honger (0-100):");
                    _pets.Add(new DigitalPet(name, energy, hunger));
                    break;

                case "2": // Lijst
                    foreach (var pet in _pets) Console.WriteLine(pet.EnergyDisplay());
                    _menu.WaitForEnter();
                    break;

                case "3": // Login
                    string search = _menu.AskString("Wie zoek je?");
                    // LINQ search
                    _currentPet = _pets.FirstOrDefault(p => p.Name.ToLower() == search.ToLower());
                    if (_currentPet == null)
                    {
                         Console.WriteLine("Niet gevonden.");
                         _menu.WaitForEnter();
                    }
                    break;

                case "Q": // Stop
                    appRunning = false;
                    break;
            }
        }
        // SITUATIE B: ACTIE MENU (Ingelogd)
        else
        {
            _menu.ShowPetMenu(_currentPet.Name, _currentPet.Energy, _currentPet.Hunger);
            string action = _menu.AskString("Actie:");

            switch (action)
            {
                case "0": // Uitloggen
                    _currentPet = null;
                    break;
            }
        }
    }
}
