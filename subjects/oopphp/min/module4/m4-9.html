Superglobals

Superglobal Wanneer gebruiken Voorbeeld
$_POST Formulierdata versturen (niet in URL). $name = $_POST['name'] ?? '';
$_GET Navigatie tussen pagina's (wel in URL). $page = $_GET['page'] ?? 'home';
$_SESSION Data bewaren tussen pagina's. $_SESSION['user'] = $user;
$_SERVER Check of formulier verzonden is. $_SERVER['REQUEST_METHOD'] === 'POST'

Formulier Workflow

// 1. Check POST request
if ($page === 'create' && $_SERVER['REQUEST_METHOD'] === 'POST') {
    // 2. Valideer input
    if (empty($_POST['name'])) {
        $smarty->assign('error', 'Naam verplicht');
        $smarty->display('form.tpl');
        exit;
    }

    // 3. Maak object
    $character = new Character($_POST['name'], $_POST['role']);

    // 4. Bewaar in sessie
    $_SESSION['characters'][] = $character;

    // 5. Toon bevestiging
    $smarty->display('created.tpl');
    exit;
}

Routing met Switch

session_start();
$page = $_GET['page'] ?? 'home';

// POST afhandelen komt VOOR switch
if ($page === 'create' && $_SERVER['REQUEST_METHOD'] === 'POST') {
    // Verwerk formulier...
}

// GET afhandelen met switch
switch ($page) {
    case 'create':
        $smarty->display('form.tpl');
        break;
    case 'list':
        $smarty->display('list.tpl');
        break;
    default:
        $smarty->display('home.tpl');
}
